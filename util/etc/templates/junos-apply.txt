        </prefix-list>

<#-- share common provider policy based on a default(MAX) logical connection BGP
        <policy-statement>

            <term>

                <from>
                    <protocol>
                        <name>bgp</name>
                    </protocol>
                    <neighbor>${bgp_p.peer_address}</neighbor>

        <policy-statement>
            <name>${policies_c.policy_name}</name>

                    </protocol>

            <term>

                <then>
                    <reject/>
                </then>
            </term>

<#if has_customer_config>

                <then>

    </firewall>

</logical-systems>
